class ScrollableFrame:
    def __init__(self, master):
        self.outer_frame = Frame(master,bg='red')
        self.canvas = Canvas(self.outer_frame, bg='orange')
        self.scrollbar = Scrollbar(self.outer_frame, orient='vertical', command=self.canvas.yview)
        self.inner_frame = Frame(self.canvas, bg='blue')
        self.inner_frame.bind('', lambda e: self.canvas.config(scrollregion=self.canvas.bbox('all')))
        self.canvas.create_window((0, 0), window=self.inner_frame, anchor='nw',width=self.canvas.winfo_reqwidth())
        self.canvas.create_window((0, 0), window=self.inner_frame, anchor='nw')
        self.canvas.config(yscrollcommand=self.scrollbar.set)
        self.canvas.pack(side='left', fill='both', expand=True)
        self.scrollbar.pack(side='right', fill='y')
